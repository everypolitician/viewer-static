require 'everypolitician'
country = Everypolitician::Index.new.country('Argentina')
house = country.legislature('Diputados')
house.popolo.terms.find_by(id: 'term/135').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
