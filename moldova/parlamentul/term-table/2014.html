require 'everypolitician'
country = Everypolitician::Index.new.country('Moldova')
house = country.legislature('Parlamentul')
house.popolo.terms.find_by(id: 'term/2014').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
