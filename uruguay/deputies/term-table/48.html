require 'everypolitician'
country = Everypolitician::Index.new.country('Uruguay')
house = country.legislature('Deputies')
house.popolo.terms.find_by(id: 'term/48').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end

Libraries also available in Python and PHP.
