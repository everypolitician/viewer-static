require 'everypolitician'
country = Everypolitician::Index.new.country('Albania')

house.popolo.terms.find_by(id: 'term/9').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
