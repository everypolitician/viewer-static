require 'everypolitician'
country = Everypolitician::Index.new.country('Malaysia')
house = country.legislature('Dewan-Rakyat')
house.popolo.terms.find_by(id: 'term/7').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
