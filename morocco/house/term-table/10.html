require 'everypolitician'
country = Everypolitician::Index.new.country('Morocco')
house = country.legislature('House')
house.popolo.terms.find_by(id: 'term/10').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
