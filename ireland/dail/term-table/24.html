require 'everypolitician'
country = Everypolitician::Index.new.country('Ireland')
house = country.legislature('Dail')
house.popolo.terms.find_by(id: 'term/24').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end

Libraries also available in Python and PHP.
