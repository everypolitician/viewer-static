require 'everypolitician'
country = Everypolitician::Index.new.country('Armenia')
house = country.legislature('Assembly')
house.popolo.terms.find_by(id: 'term/6').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
