require 'everypolitician'
country = Everypolitician::Index.new.country('Philippines')
house = country.legislature('House')
house.popolo.terms.find_by(id: 'term/16').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end

Libraries also available in Python and PHP.
