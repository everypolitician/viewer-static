require 'everypolitician'
country = Everypolitician::Index.new.country('Serbia')
house = country.legislature('National-Assembly')
house.popolo.terms.find_by(id: 'term/11').memberships.each do |membership|
  puts "#{membership.person.name} (#{membership.party.name}): #{membership.person.gender}"
end
